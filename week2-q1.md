def total_steps():
    footsteps = 0
    for day in range(7):
        daily_input = int(input(f"Enter the footsteps taken on day {day + 1}: "))
        footsteps += daily_input
    return footsteps

total = total_steps()
print("Total footsteps for the week:" total)
